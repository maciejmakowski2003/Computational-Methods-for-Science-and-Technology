function test()
    open("time_test.csv", "w") do file
        write(file, "Size;Scalar_Time;Matrix_Time\n")
        for size = 100:300:1601
            for i = 1:11
                Random.seed!(size+i)
                x = rand(0:9, size)
                y = rand(0:9, size)
                m = rand(0:9, (size, size))
                c = rand(0:9, size) 

                scalar_time = @elapsed dot(x,y)
                matrix_time = @elapsed m * c

                if i > 1 
                    write(file, "$size;$scalar_time;$matrix_time\n")
                end
            end
        end
    end
end
